\section{Flusso di Fase}%
\begin{ex}[Sul flusso di fase]
    Verificare la validità delle 3 proprietà der il flusso di fase:
    \[
        \varphi_t = 
	\begin{pmatrix} 
	    e^{-\Gamma t} & 0 \\
	    0 & e^{\Gamma t} 
        \end{pmatrix} 
    .\] 
    Per il seguente sistema dinamico:
    \[
    \begin{dcases}
    \dot{x}_1 = - \Gamma x_1\\
    \dot{x}_2= \Gamma x_2
    \end{dcases}
    \]
\end{ex}
